    <script>

        function calculateGrade() {

            // Get user input

            const d2lScore = parseFloat(document.getElementById('d2lScore').value);

            const d2lTotal = parseFloat(document.getElementById('d2lTotal').value);

            const weight = parseFloat(document.getElementById('weight').value) / 100;

            const finalWeight = parseFloat(document.getElementById('finalWeight').value) / 100;

            const passingGrade = parseFloat(document.getElementById('passingGrade').value);


            // Calculate current grade

            const weightedGrade = (d2lScore / d2lTotal) * weight * 100;


            // Calculate remaining grade needed

            const remainingWeight = 1 - weight;

            const gradeNeededInFinal = (passingGrade - weightedGrade) / finalWeight;


            // Display the result

            const result = document.getElementById('result');

            if (gradeNeededInFinal > 100) {

                result.textContent = "Unfortunately, it is not possible to pass the course.";

            } else if (gradeNeededInFinal <= 0) {

                result.textContent = "Congratulations! You have already passed the course.";

            } else {

                result.textContent = `You need to score at least ${gradeNeededInFinal.toFixed(2)}% on the final exam to pass.`;

            }

        }

    </script>
